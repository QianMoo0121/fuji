\LevelFour{How it works?}
When a player use the command \cmd{/msg} or \cmd{/tell}, we track the usage, and allow the target player to use \cmd{/reply} to send message to the player recently meg he.

\begin{tips}{Create a command alias like \cmd{/r}}
    You can use~\nameref{ch:command_alias} module to create a command alias from \cmd{/r} to \cmd{/reply}.
\end{tips}

\LevelThree{tppos}
\fcmd{/tppos}


\LevelFour{Example}
\begin{example}{Teleport to offline location of a player}
    \cmd{/tppos offline \player}
\end{example}

\begin{example}{Specify another player as the target player}
    \cmd{/tppos --targetPlayer \player}
\end{example}

\begin{example}{Use fixed teleport mode}
    When any of the following args is specified, the \cmd{/tppos} command will do a \ttt{fixed teleport}.\\
    \cmd{/tppos --x 32 --y 64 --z 128 --yaw 60 --pitch 90 --dimension minecraft:overworld}
\end{example}

\begin{example}{Use random teleport mode}
    The \cmd{/tppos} command will do a \ttt{random teleport} if not doing a \ttt{fixed teleport}.\\
    \cmd{/tppos --centerX 0 --centerZ 0 --circle true --minRange 512 --maxRange 2048 --minY 32 --maxY 128 --maxTryTimes 8 --dimension minecraft:overworld}
\end{example}
