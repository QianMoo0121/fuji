\LevelOne{color}

\LevelTwo{Purpose}
This module provides colorize for things.

\LevelTwo{Sub-Module}
\LevelThree{sign}
\LevelFour{Purpose}
Colorize the sign.

\LevelFour{Configuration}
\begin{Configuration}
    \item[requires\_corresponding\_permission\_to\_use\_style\_tag]{By default, any player can use all style tags. Enable this options requires the player to has corresponding permission to use style tag. e.g. \str{fuji.style.sign.<style-tag>}}
\end{Configuration}

\LevelFour{Example}
\begin{example}{Write colorful text in sign}
    <red>This is the first line\\
    <rb>The second line\\
    <bold>The third line\\
    <i>The last line
\end{example}
