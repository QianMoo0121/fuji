\LevelThree{rewrite}\label{ch:chat_rewrite}
\LevelFour{Purpose}
Allow you to define \ttt{regex} to replace \ttt{chat message} sent by players.

\LevelFour{Example}
\begin{example}{Rewrite the \str{item} into \str{[item]} as a shortcut.}
    \begin{jsoncode}
    {
        "regex": "(?<=^|\\s)item(?=\\s|$)",
        "replacement": "[item]"
    }
    \end{jsoncode}
\end{example}

\begin{example}{Rewrite badwords as a message filter.}
    \begin{jsoncode}
    {
        "regex": "(bad-word)|(bad-word-2)|(bad-word-3)",
        "replacement": "<st>***</st>"
    }
    \end{jsoncode}
\end{example}
