\LevelThree{json}

\LevelFour{Purpose}
Provides a unified json editor.

\LevelFour{Command}
\fcmd{/json}

\LevelFour{Exaxmple}
\begin{example}{Read a key}
    \cmd{/json read "config/fuji/config.json" "\$.core.quartz.logger\_level"}
\end{example}

\begin{example}{List keys}
    \cmd{/json read "config/fuji/config.json" "\$.modules.keys()"}
\end{example}

\begin{example}{Set a key}
    \cmd{/json write "config/fuji/config.json" "\$.core.quartz.logger\_level" STRING INFO"}
\end{example}
