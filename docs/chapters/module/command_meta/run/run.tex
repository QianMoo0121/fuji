\LevelThree{run}

\LevelFour{Purpose}
This module provides \cmd{/run} command, which can run a command with context.

\LevelFour{Command}
\fcmd{/run}

\LevelFour{Example}
\begin{example}{Give random diamonds to online players}
    \cmd{/run as console give @a minecraft:diamond \ph{fuji:random 8 32}}
\end{example}

\begin{example}{Give online players random diamonds}
    \cmd{/run as console foreach give \ph{fuji:escape player:name} minecraft:diamond \ph{fuji:escape fuji:random 8 32 1}}
\end{example}

\begin{example}{Execute a command as a player}
   \cmd{/run as player \player back}
\end{example}

\begin{example}{Execute a command as fake-op}
   \cmd{/run as fake-op \player give \ph{player:name} minecraft:apple 1}
\end{example}
