\LevelOne{skin}
\LevelTwo{Purpose}
This module provides skin customization.

\LevelTwo{Command}
\fcmd{/skin}

\LevelTwo{Configuration}
\begin{Configuration}
    \item[default\_skins]{
        The default skin used for player who has no skin set.
    }
\end{Configuration}

\LevelTwo{Example}

\begin{example}{Set a skin from mojang}
    \cmd{/skin set mojang dream}
\end{example}

\begin{example}{Set a skin from url}
    \cmd{/skin set web slim "https://s.namemc.com/i/bd53d152d0cd91d0.png"}
\end{example}

\begin{example}{Use default skins}
    \cmd{/skin use-default-skins}
\end{example}

\begin{example}{Use online skin}
    \cmd{/skin use-online-skin}
\end{example}
