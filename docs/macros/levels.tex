%! suppress = DiscouragedUseOfDef
%! suppress = Makeatletter
\usepackage{titlesec}

% set max depth
\setcounter{secnumdepth}{10}
\setcounter{tocdepth}{10}

% level 0
\newcommand{\LevelZero}[1]{\chapter{#1}}

% level 1
\newcommand{\LevelOne}[1]{\section{#1}}

% level 2
\newcommand{\LevelTwo}[1]{\subsection{#1}}

% level 3
\newcommand{\LevelThree}[1]{\subsubsection{#1}}

% level 4
\makeatletter
\let\stdl@paragraph\l@paragraph
\renewcommand*{\l@paragraph}[2]{\stdl@paragraph{\hspace{10pt}#1}{#2}}
\makeatother
\newcommand{\LevelFour}[1]{\paragraph{#1}\mbox{}\\}

% level 5
\makeatletter
\let\stdl@subparagraph\l@subparagraph
\renewcommand*{\l@subparagraph}[2]{\stdl@subparagraph{\hspace{40pt}#1}{#2}}
\makeatother

\titlespacing*{\subparagraph}{15pt}{0ex plus 1ex minus .2ex}{1em}

\newcommand{\LevelFive}[1]{\subparagraph{#1}\mbox{}\\}
